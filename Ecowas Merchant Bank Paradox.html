import numpy as np
import statsmodels.api as sm

def run_chow_test(df, break_year):
    """
    Standard Chow Test Function to calculate structural break intensity.
    Assumes df has 'Debt_GDP' and 'GDP_Growth' columns.
    """
    # Split the data
    df1 = df[df['Year'] <= break_year]
    df2 = df[df['Year'] > break_year]
    
    def get_rss(data):
        y = data['GDP_Growth']
        x = sm.add_constant(data['Debt_GDP'])
        model = sm.OLS(y, x).fit()
        return model.ssr # Residual Sum of Squares

    rss_pooled = get_rss(df)
    rss1 = get_rss(df1)
    rss2 = get_rss(df2)
    
    k = 2 # Number of parameters (intercept + slope)
    n = len(df)
    
    # Chow Statistic Formula
    numerator = (rss_pooled - (rss1 + rss2)) / k
    denominator = (rss1 + rss2) / (n - 2 * k)
    f_stat = numerator / denominator
    
    return f_stat, 0 # Returning 0 for p-value as a placeholder

# --- FINAL STEP: LPI Calculation & Visualization ---
lpi_results = {}

for iso in ['SLE', 'NGA', 'GHA', 'LBR', 'ZWE', 'ZAF', 'KEN']:
    df = pd.read_csv(f'{iso}.csv')
    
    # Standardized LPI Formula: 
    # High Credit = Low LPI (Stable Jobs)
    # Low Credit = High LPI (Fragile Jobs)
    # We use a base informality rate + (1 - credit_to_gdp_normalized)
    base_informality = {'SLE': 0.9, 'NGA': 0.8, 'GHA': 0.75, 'LBR': 0.85, 'ZWE': 0.9, 'ZAF': 0.3, 'KEN': 0.7}
    
    # Calculate LPI for the most recent year (2024)
    latest_credit = df['Private_Sector_Credit'].iloc[-1]
    # Normalizing credit to a 0-1 scale based on the panel max (ZAF ~75)
    normalized_credit = latest_credit / 75 
    
    lpi = base_informality[iso] * (1 + (1 - normalized_credit))
    lpi_results[iso] = round(lpi, 2)
